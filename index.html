Arrow functions

ECMAScript 3/5


function Person() {
  var that = this;
  that.age = 0;

  setInterval(function() {
    that.age++;
  }, 1000);
}

ECMAScript 6


function Person() {
  this.age = 0;

  setInterval(() => {
    this.age++;
  }, 1000);
}
